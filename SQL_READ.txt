//find the sum of the salaries for employees 
//in department 10 along with the sum of their bonuses
-------
select deptno,
  sum(distinct sal) as total_sal,
  sum(bonus) as total_bonus
from (
   select e.empno, 
          e.ename, 
          e.sal, 
          e.deptno, 
          e.sal*case when eb.type = 1 then .1
 		     when eb.type = 2 then .2
 		     else .3 end as bonus
   from emp e, emp_bonus eb
   where e.empno = eb.empno and e.deptno = 10
) as x group by deptno
